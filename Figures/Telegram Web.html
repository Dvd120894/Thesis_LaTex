function [O1, O2, O3] = fftplot(Fs, data)
% Function that plot Power Spectral Density
% Input: Sampling Frequency, sperimental Data
% Output: Power Spectral Density, Related Frequency Axis

% FFT of data
FT = fft(data)*2/length(data);
N = length(FT);
% PSD of data
S = (abs(FT).^2)/N;
% Related Frequency Axis
Freq = (0:Fs/N:Fs-Fs/N);
% One-sided plot
Freq = Freq(1:N/2);
% One-sided PSD and FFT
FT = FT(1:N/2);
S = S(1:N/2);
% Outputs
O1 = FT;
O2 = S;
O3 = Freq;
end
